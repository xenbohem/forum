import { useState, useEffect } from 'react';
import { supabase } from './supabaseClient'; // Supabase client

function NexusChat({ username }) {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');

  useEffect(() => {
    // Initial greeting
    setMessages([{ text: `Welcome, Agent ${username}. The Nexus awaits your command. Suggest: Explore the Mantra Vault for daily wisdom.`, sender: 'Nexus AI' }]);
  }, [username]);

  const sendMessage = async () => {
    // Call Supabase Edge Function for AI response
    const { data } = await supabase.functions.invoke('nexus-ai', { body: { message: input, username } });
    setMessages([...messages, { text: input, sender: username }, { text: data.response, sender: 'Nexus AI' }]);
    setInput('');
  };

  return (
    <div className="bg-black text-cyan-400 p-4 font-mono">
      {messages.map((msg, i) => (
        <p key={i} className={msg.sender === 'Nexus AI' ? 'text-purple-400' : ''}>
          {msg.sender}: {msg.text}
        </p>
      ))}
      <input value={input} onChange={(e) => setInput(e.target.value)} className="bg-indigo-900 text-white" />
      <button onClick={sendMessage} className="bg-cyan-600">Transmit</button>
    </div>
  );
}

// Initialize Supabase
const supabaseUrl = 'YOUR_SUPABASE_URL'; // Replace with your Supabase project URL
const supabaseKey = 'YOUR_SUPABASE_ANON_KEY'; // Replace with your anon key
const supabase = window.supabase.createClient(supabaseUrl, supabaseKey);

// Global variables
let username = 'Agent'; // Set via auth later
let messages = [];

// Boot sequence
window.onload = () => {
    setTimeout(() => {
        document.getElementById('boot-sequence').innerHTML = `Nexus Online. Welcome, ${username}.`;
        addMessage(`Welcome, ${username}. The Nexus awaits. Suggest: Save a mantra in your Vault.`, 'Nexus AI');
    }, 2000);
};

// Chat functionality
document.getElementById('send-btn').addEventListener('click', async () => {
    const input = document.getElementById('chat-input').value;
    if (!input) return;
    addMessage(input, username);
    document.getElementById('chat-input').value = '';

    // Call Supabase Edge Function for AI response
    try {
        const { data } = await supabase.functions.invoke('nexus-ai', { body: { message: input, username } });
        addMessage(data.response, 'Nexus AI');
    } catch (error) {
        addMessage('Error: Unable to connect to Nexus AI.', 'Nexus AI');
    }
});

function addMessage(text, sender) {
    messages.push({ text, sender });
    const msgDiv = document.getElementById('messages');
    msgDiv.innerHTML += `<p style="color: ${sender === 'Nexus AI' ? '#800080' : '#00FFFF'}">${sender}: ${text}</p>`;
    msgDiv.scrollTop = msgDiv.scrollHeight;
}

// Note-saving (localStorage + Supabase sync)
document.getElementById('save-note').addEventListener('click', async () => {
    const note = document.getElementById('note-input').value;
    if (!note) return;

    // Save to localStorage
    let notes = JSON.parse(localStorage.getItem('nexus-notes') || '[]');
    notes.push({ text: note, timestamp: new Date().toISOString() });
    localStorage.setItem('nexus-notes', JSON.stringify(notes));

    // Sync to Supabase (assuming a 'notes' table)
    await supabase.from('notes').insert([{ user_id: 'current-user-id', content: note }]); // Add auth for user_id

    loadNotes();
    document.getElementById('note-input').value = '';
});

function loadNotes() {
    const notes = JSON.parse(localStorage.getItem('nexus-notes') || '[]');
    const list = document.getElementById('notes-list');
    list.innerHTML = notes.map(note => `<li>${note.text} (${new Date(note.timestamp).toLocaleDateString()})</li>`).join('');
}

loadNotes(); // Load on page load
